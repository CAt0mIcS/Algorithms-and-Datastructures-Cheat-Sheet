\subsection{Sortieralgorithmen}

\begin{sectionbox}
    \subsubsection{SelectionSort}
    Sucht das kleinste Element in der Liste, fügt es am Anfang einer neuen Liste ein und löscht es aus der alten Liste.
    Wiederholt, bis alte Liste leer ist $\rightarrow O(n^2)$
\end{sectionbox}

\begin{sectionbox}
    \subsubsection{InsertionSort}
    \begin{enumerate}
        \item Wähle num = input[i]
        \item Vergleiche alle Indices j $<$ i und schiebe num nach input[j], wenn input[j] $<$ num
    \end{enumerate}
    \begin{tabularx}{\linewidth}{@{}>{\bfseries}l@{\hspace{.5em}}|X@{}}
        Laufzeit & \textbf{Best case}: $O(n)$ falls Input sortiert                                       \\
                 & \textbf{Worst case}: $O(n^2)$ falls Input absteigend sortiert                         \\
                 & $O(hn)$ falls ein Element höchstens $h$ Elemente von seiner Zielposition entfernt ist
    \end{tabularx}
\end{sectionbox}

\begin{sectionbox}
    \subsubsection{MergeSort}
    Liste wird rekursiv halbiert, bis zwei Listen mit je einem Element entstehen. Diese werden jeweils sortiert und mit anderen Teilergebnissen zusammengefügt
    \begin{tabularx}{\linewidth}{@{}>{\bfseries}l@{\hspace{.5em}}|X@{}}
        Laufzeit & \textbf{Worst case}: $T(n) = 2T(\frac{n}{2}) + O(n) \Rightarrow O(n\log_2 n)$
    \end{tabularx}

    \begin{sectionbox}
        \parbox{4cm}{
            \t{mergeSort(List* a):\\
                if (a$\rightarrow$length() $<$= 1)\\
                \hspace*{1em} return a;\\
                List* b = a$\rightarrow$half();\\
                a = mergeSort(a);\\
                b = mergeSort(b);\\
                return merge(a,b);\\
                \\
                half():\\
                Node* ptr = head;\\
                for (0 to length/2-2)\\
                \hspace*{1em} ptr = ptr$\rightarrow$next;\\
                List* L = new (ptr$\rightarrow$next);\\
                ptr$\rightarrow$next = NULL;\\
                return L;
            }}\hspace{-0.6cm}
        \parbox{2.9cm}{
            \t{
                merge(a, b):\\
                if (a$\rightarrow$empty())\\
                \hspace*{1em} delete a; return b;\\
                if (b$\rightarrow$empty())\\
                \hspace*{1em} delete b; return a;\\
                List* h;\\
                if a[0] $<$ b[0]\\
                \hspace*{1em} h = a; \\
                else h = b;\\
                int d = h$\rightarrow$delFront();\\
                List L = merge(a,b);\\
                L$\rightarrow$insertFront(d);\\
                return L;\\
            }}
    \end{sectionbox}

\end{sectionbox}

\begin{sectionbox}
    \subsubsection{QuickSort}
    \begin{enumerate}
        \item Wähle ein Pivotelement A[p]
        \item Wähle Index $i,j=\text{start}$. Gehe durch die gesamte Liste (überspringe Pivot).
              \begin{itemize}
                  \item Falls A[j] $>$ A[p] $\rightarrow$ j++
                  \item Falls A[j] $<$ A[p] $\rightarrow$ swap(i, j), i++, j++
              \end{itemize}
              % \item Bewege alle Elemente $<$ A[p] nach links von A[p]
              % \item Bewege alle Elemente $>$ A[p] nach rechts von A[p]
        \item Teile Liste in zwei Teile, jeweils links und rechts von A[p] (ohne A[p])
    \end{enumerate}
    \begin{tabularx}{\linewidth}{@{}>{\bfseries}l@{\hspace{.5em}}|X@{}}
        Laufzeit & \textbf{Worst case}: $T(n-1) + O(n) \rightarrow O(n^2)$ falls als Pivotelement immer das größte Element gewählt wird (sehr selten) \\
                 & \textbf{Average}: $T(\frac{n}{2}) + O(n) \rightarrow O(n\log n)$
    \end{tabularx}

    \begin{sectionbox}
        \parbox{7cm}{\t{quickSort(A, left, right): \\
                if(right $<$= left) return \\
                pivotIdx = partition(A, left, right) \\
                quickSort(A, left, pivotIdx - 1) \\
                quickSort(A, pivotIdx + 1, right)
            }}

        \vspace{3mm}

        \parbox{4cm}{\t{partition(A, left, right): \\
                i = left \\
                pivot = A[r] \\
                for(j = left; j $<$= right; j++) \\
                \hspace*{1em} if(A[j] $<$= pivot) \\
                \hspace*{2em} swap(A, i, j) \\
                \hspace*{2em} i++ \\
                return i - 1
            }}
    \end{sectionbox}

\end{sectionbox}
